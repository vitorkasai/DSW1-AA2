
	console.log("Arquivo JavaScript carregado!");
	document.addEventListener('DOMContentLoaded', function () {
		console.log("Documento carregado!");
		var cidadeSelect = document.getElementById('selectCidade');
		var xhr = new XMLHttpRequest();
		
		xhr.open('GET', '/locadoras/cidades', true); // Mapeie essa URL no seu controlador para retornar o conte√∫do do arquivo de cidades
		xhr.onreadystatechange = function() {
			if (xhr.readyState === 4 && xhr.status === 200) {
				var cidades = xhr.responseText.split('\n');
				for (var i = 0; i < cidades.length; i++) {
					var option = document.createElement('option');
					option.text = cidades[i];
					cidadeSelect.add(option);
				}
			}
		};
		xhr.send();

		cidadeSelect.addEventListener('change', function() {
			var cidadeSelecionada = cidadeSelect.value;
			var locadoras = document.querySelectorAll('[data-cidade]');
	
			for (var i = 0; i < locadoras.length; i++) {
				var locadora = locadoras[i];
				var cidadeLocadora = locadora.getAttribute('data-cidade');
	
				if (cidadeSelecionada === 'allCidades' || cidadeSelecionada === cidadeLocadora) {
					locadora.style.display = 'table-row';
				} else {
					locadora.style.display = 'none';
				}
			}
		});
	});
